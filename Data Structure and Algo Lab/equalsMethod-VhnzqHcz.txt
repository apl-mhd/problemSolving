public class Person {

    String name;
    String ID;
    int age;

    public Person(String name, String ID, int age) {
        this.name = name;
        this.ID = ID;
        this.age = age;
    }

    public boolean equals(Object o){

        if(this == o)
            return true;

        if(o.getClass() !=this.getClass()){

            return false;
        }

        Person p =  (Person)o;

        return name.equals(p.name);
    }

    public String toString(){


        return  name+" "+ID+" "+age;
    }

}
